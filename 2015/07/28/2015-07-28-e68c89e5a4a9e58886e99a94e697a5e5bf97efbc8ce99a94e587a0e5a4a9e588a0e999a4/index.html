按天分隔日志，隔几天删除

ErrorLog “| /usr/local/apache/bin/rotatelogs /home/logs/www/%Y%m%d_errorlog 86400 480″
CustomLog “| /usr/local/apache/bin/rotatelogs /home/logs/www/%Y%m_%d_access_log 86400 480″ common
0 1 /usr/bin/find /alidata/logs/www -name \access_log -mtime +2 | xargs rm -rf
